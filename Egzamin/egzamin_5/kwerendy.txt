1.SELECT wodowskazy.nazwa,wodowskazy.rzeka,wodowskazy.stanAlarmowy FROM wodowskazy;
2.SELECT wodowskazy.nazwa,wodowskazy.rzeka,wodowskazy.stanOstrzegawczy,wodowskazy.stanAlarmowy,pomiary.stanWody FROM wodowskazy JOIN pomiary ON wodowskazy.id=pomiary.wodowskazy_id WHERE pomiary.dataPomiaru='2022-05-05';
3.SELECT wodowskazy.nazwa,wodowskazy.rzeka,wodowskazy.stanOstrzegawczy,wodowskazy.stanAlarmowy,pomiary.stanWody FROM wodowskazy JOIN pomiary ON wodowskazy.id=pomiary.wodowskazy_id WHERE pomiary.dataPomiaru='2022-05-05' AND pomiary.stanWody>wodowskazy.stanAlarmowy;
4.SELECT pomiary.dataPomiaru,AVG(pomiary.stanWody) FROM pomiary GROUP BY pomiary.dataPomiaru;
