2.SELECT DISTINCT(smok.pochodzenie) FROM smok ORDER BY smok.pochodzenie ASC;
3.SELECT parada.rok,AVG(smok.dlugosc) AS 'Średnia długość' FROM smok JOIN udzial ON smok.id=udzial.id_smok JOIN parada ON udzial.id_parada=parada.id WHERE parada.rok > 2005 GROUP BY parada.rok;
